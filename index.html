How to install:

<style>
.mosaicflow__column {
	float:left;
	}
.item img {
	display:block;
	width:100%;
	height:auto;
	}
</style>

<div class="clearfix mosaicflow">
	<div class="post">
		<img width="300" height="150" src="1.jpg" alt="">
	</div>
	<div class="post">
		<img width="150" height="300" src="2.jpg" alt="">
	</div>
	…
</div>

<script src="http://ajax.googleapis.com/ajax/libs/jquery/1.9.0/jquery.min.js"></script>
<script src="jquery.mosaicflow.js"></script>
<script>
$('.mosaicflow').mosaicflow();
</script>
